bool isLastMessageLeft (int index, dynamic listMessage, String id)

Check if last message is sent by peer Returns boolean true/false

Implementation 

static bool isLastMessageLeft(int index, var listMessage, String id) {
  if ((index > 0 &&
          listMessage != null &&
          listMessage[index - 1]['idFrom'] == id) ||
      index == 0) {
    return true;
  } else {
    return false;
  }
}
